select multas.data, veiculos.placa, multas.valor, multas.infracao, multas.nivel, multas.local, reservas.funcionario, multas.pdf_multa
from multas
join reservas
on condicao = 'reservado' and multas.data > reservas.data_saida and multas.id_veiculo = reservas.id_veiculo
or multas.data between reservas.data_saida and reservas.data_retorno and multas.id_veiculo = reservas.id_veiculo
join veiculos on veiculos.id_veiculo = multas.id_veiculo and reservas.tipo_reserva = 'comum';
